BASE_2_8_To_10
	WHILE True DO
		GET button // รับ่าทางคีย์บอร์ด
		IF button = 'B' or button = 'b' THEN // ถ้าคีย์บอร์ดมีค่า B หรือ b
			GET Base2 // รับค่า Base2
			SET Total_Base2 = 0 // ตั้งค่า Total_Base2 เป็น 0
			SET Len_Base2 = Base2 // ตั้งค่า Len_Base2 เป็น Base2
			SET Multiply = 0
			FOR x = 1 to Base2 DO // วนลูปตั้งต้นจาก 1 ไปจนถึง Base2
				Len_Base2 = Len_Base2 - 1 // ลดค่า Len_Base2 ลง 1
				Multiply = 2**Len_Base2
				Total_Base2 = Total_Base2 + (int(Base2(x)) * Multiply)
			ENDFOR
			DISPLAY Total_Base2

		ELSE IF button = 'O' or button = 'o' THEN
			GET Base8
			SET Total_Base8 = 0
			SET Len_Base8 = Base8
			SET Multiply = 0
			FOR y = 1 to Base8 DO
				Len_Base8 = Len_Base8 - 1
				Multiply = 8**Len_Base8
				Total_Base8 = Total_Base8 + (int(Base8(y)) * Multiply)
			ENDFOR
			DISPLAY Total_Base8
		
		ELSE
			break
		ENDIF
		
	ENDWHILE
END
